\section{Application Roles}

\subsection{Training Application (Local)}
\begin{itemize}
	\item Installed and run on your personal computer
	\item Processes historical sales data
	\item Trains forecasting models using selected algorithms
	\item Saves models to disk for later use
\end{itemize}

\subsection{Prediction Application (Cloud)}
\begin{itemize}
	\item Accessible through any web browser
	\item No installation required
	\item Uses trained models to generate forecasts
	\item Provides interactive visualization and data export
\end{itemize}


\section{Key Features}

\subsection{Model Training}
\begin{itemize}
	\item Support for two modeling algorithms:
	\begin{itemize}
		\item Auto ARIMA (automatic parameter selection)
		\item Exponential Smoothing (Holt-Winters method)
	\end{itemize}
	\item Customizable hyperparameters
	\item Performance evaluation using WMAE metric
\end{itemize}

\subsection{Prediction Generation}
\begin{itemize}
	\item 4-week forecast generation
	\item Interactive visualizations
	\item Multiple export formats (CSV, JSON)
	\item Summary statistics
\end{itemize}

\section{System Requirements}

\subsection{Training App}
\begin{itemize}
	\item Windows, macOS, or Linux
	\item Python 3.8+
	\item Internet connection for initial setup
	\item 2GB RAM minimum
\end{itemize}

\subsection{Prediction App}
\begin{itemize}
	\item Modern web browser (Chrome, Firefox, Safari, Edge)
	\item Stable internet connection
	\item No additional software required
\end{itemize}
